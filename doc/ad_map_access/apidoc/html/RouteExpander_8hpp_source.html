187 template <class RoutingCostData>
188 void RouteExpander<RoutingCostData>::expandNeighbors(
189  typename RouteExpander<RoutingCostData>::RoutingPoint const &origin)
190 {
191  if (origin.first.direction == RoutingDirection::DONT_CARE)
192  {
193  throw std::runtime_error("RouteExpander::ExpandNeighbors invalid routing direction!");
194  }
195 
196  lane::Lane::ConstPtr lane = lane::getLanePtr(origin.first.point.lane_id);
197  if (lane)
198  {
199  if ( // lane has to be routable to expand
200  isRouteable(*lane)
201  && ( // max distance and max duration are not yet reached
202  ((origin.second.routeDistance < mMaxDistance) && (origin.second.routeDuration < mMaxDuration))
203  // or we are within an intersection
204  || lane::isLanePartOfAnIntersection(*lane)))
205  {
206  expandSameLaneNeighbors(lane, origin);
207  expandLongitudinalNeighbors(lane, origin);
208  expandLateralNeighbors(lane, origin);
209  }
210  }
211  else
212  {
213  throw std::runtime_error("RouteExpander::ExpandNeighbors No lane!");
214  }
215 }

217 template <class RoutingCostData>
218 typename RouteExpander<RoutingCostData>::RoutingPoint
219 RouteExpander<RoutingCostData>::createNeighbor(lane::Lane::ConstPtr originLane,
220  RoutingPoint const &origin,
221  lane::Lane::ConstPtr neighborLane,
222  RoutingParaPoint neighborRoutingParaPoint)
223 {
224  RoutingPoint neighbor;
225  neighbor.first = neighborRoutingParaPoint;
226 
227  physics::Distance neighborDistance{0.};
228  physics::Duration neighborDuration{0.};
229  point::ECEFPoint pt_origin
230  = getParametricPoint(*originLane, origin.first.point.parametric_offset, physics::ParametricValue(0.5));
231  point::ECEFPoint pt_neighbor
232  = getParametricPoint(*neighborLane, neighbor.first.point.parametric_offset, physics::ParametricValue(0.5));
233  neighborDistance = point::distance(pt_neighbor, pt_origin);
234  physics::ParametricRange drivingRange;
235  if (origin.first.point.parametric_offset < neighbor.first.point.parametric_offset)
236  {
237  drivingRange.minimum = origin.first.point.parametric_offset;
238  drivingRange.maximum = neighbor.first.point.parametric_offset;
239  }
240  else
241  {
242  drivingRange.minimum = neighbor.first.point.parametric_offset;
243  drivingRange.maximum = origin.first.point.parametric_offset;
244  }
245 
246  if (originLane == neighborLane)
247  {
248  neighborDuration = getDuration(*originLane, drivingRange);
249  }
250  else
251  {
252  auto const maxSpeed = std::max(getMaxSpeed(*originLane, drivingRange), cMinimumSpeed);
253  neighborDuration = neighborDistance / maxSpeed;
254  }
255  neighborDistance = std::max(neighborDistance, cMinimumNeighborDistance);
256  neighborDuration = std::max(neighborDuration, cMinimumNeighborDuration);
257  neighbor.second.routeDistance = origin.second.routeDistance + neighborDistance;
258  neighbor.second.routeDuration = origin.second.routeDuration + neighborDuration;
259  return neighbor;
260 }

274 template <class RoutingCostData>
275 void RouteExpander<RoutingCostData>::expandSameLaneNeighbors(
276  lane::Lane::ConstPtr lane, typename RouteExpander<RoutingCostData>::RoutingPoint const &origin)
277 {
278  if ((lane->id == getDest().lane_id)
279  && ((isPositiveMovement(lane, origin) && (origin.first.point.parametric_offset <= getDest().parametric_offset))
280  || (isNegativeMovement(lane, origin)
281  && (origin.first.point.parametric_offset >= getDest().parametric_offset))))
282  {
283  // approaching the destination from valid side
284  auto const neighbor = createNeighbor(lane, origin, lane, getRoutingDest());
285  addNeighbor(lane, origin, lane, neighbor, ExpandReason::Destination);
286  }
287  if (isPositiveMovement(lane, origin) && !isEnd(origin))
288  {
289  auto const neighbor = createNeighbor(
290  lane, origin, lane, createRoutingParaPoint(lane->id, physics::ParametricValue(1.), origin.first.direction));
291  addNeighbor(lane, origin, lane, neighbor, ExpandReason::SameLaneNeighbor);
292  }
293  if (isNegativeMovement(lane, origin) && !isStart(origin))
294  {
295  auto const neighbor = createNeighbor(
296  lane, origin, lane, createRoutingParaPoint(lane->id, physics::ParametricValue(0.), origin.first.direction));
297  addNeighbor(lane, origin, lane, neighbor, ExpandReason::SameLaneNeighbor);
298  }

301 template <class RoutingCostData>
302 void RouteExpander<RoutingCostData>::expandLongitudinalNeighbors(
303  lane::Lane::ConstPtr lane, typename RouteExpander<RoutingCostData>::RoutingPoint const &origin)
304 {
305  lane::ContactLaneList contact_lanes;
306  if (isEnd(origin) && isPositiveMovement(lane, origin))
307  {
308  contact_lanes = lane::getContactLanes(*lane, lane::ContactLocation::SUCCESSOR);
309  }
310  else if (isStart(origin) && isNegativeMovement(lane, origin))
311  {
312  contact_lanes = lane::getContactLanes(*lane, lane::ContactLocation::PREDECESSOR);
313  }
314  for (auto contactLane : contact_lanes)
315  {
316  if (!isLaneRelevantForExpansion(contactLane.to_lane))
317  {
318  continue;
319  }
320  lane::Lane::ConstPtr otherLane = lane::getLanePtr(contactLane.to_lane);
321  if (otherLane)
322  {
323  if (isRouteable(*otherLane))
324  {
325  lane::ContactLocation otherToLane = getContactLocation(*otherLane, lane->id);
326  if (otherToLane == lane::ContactLocation::SUCCESSOR)
327  {
328  auto routingDirection = RoutingDirection::NEGATIVE;
329  auto const neighbor = createLongitudinalNeighbor(
330  origin, createRoutingParaPoint(otherLane->id, physics::ParametricValue(1.), routingDirection));
331 
332  addNeighbor(lane, origin, otherLane, neighbor, ExpandReason::LongitudinalNeighbor);
333  }
334  else if (otherToLane == lane::ContactLocation::PREDECESSOR)
335  {
336  auto routingDirection = RoutingDirection::POSITIVE;
337  auto const neighbor = createLongitudinalNeighbor(
338  origin, createRoutingParaPoint(otherLane->id, physics::ParametricValue(0.), routingDirection));
339 
340  addNeighbor(lane, origin, otherLane, neighbor, ExpandReason::LongitudinalNeighbor);
341  }
342  else
343  {
344  throw std::runtime_error("Other lane neither SUCCESSOR not PREDECESSOR!");
345  }
346  }
347  }
348  else
349  {
350  throw std::runtime_error("No other lane!");
351  }
352  }
353 }
