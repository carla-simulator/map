160 void RouteExpander<RoutingCostData>::expandNeighbors(
161  typename RouteExpander<RoutingCostData>::RoutingPoint const &origin)
162 {
163  lane::Lane::ConstPtr lane = lane::getLanePtr(origin.first.point.laneId);
164  if (lane)
165  {
166  if ( // lane has to be routable to expand
167  isRouteable(*lane)
168  && ( // max distance and max duration are not yet reached
169  ((origin.second.routeDistance < mMaxDistance) && (origin.second.routeDuration < mMaxDuration))
170  // or we are within an intersection
171  || lane::isLanePartOfAnIntersection(*lane)))
172  {
173  expandSameLaneNeighbors(lane, origin);
174  expandLongitudinalNeighbors(lane, origin);
175  expandLateralNeighbors(lane, origin);
176  }
177  }
178  else
179  {
180  throw std::runtime_error("RouteExpander::ExpandNeighbors No lane!");
181  }
182 }

184 template <class RoutingCostData>
185 typename RouteExpander<RoutingCostData>::RoutingPoint
186 RouteExpander<RoutingCostData>::createNeighbor(lane::Lane::ConstPtr originLane,
187  RoutingPoint const &origin,
188  lane::Lane::ConstPtr neighborLane,
189  RoutingParaPoint neighborRoutingParaPoint)
190 {
191  RoutingPoint neighbor;
192  neighbor.first = neighborRoutingParaPoint;
193 
194  physics::Distance neighborDistance{0.};
195  physics::Duration neighborDuration{0.};
196  point::ECEFPoint pt_origin
197  = getParametricPoint(*originLane, origin.first.point.parametricOffset, physics::ParametricValue(0.5));
198  point::ECEFPoint pt_neighbor
199  = getParametricPoint(*neighborLane, neighbor.first.point.parametricOffset, physics::ParametricValue(0.5));
200  neighborDistance = point::distance(pt_neighbor, pt_origin);
201  physics::ParametricRange drivingRange;
202  if (origin.first.point.parametricOffset < neighbor.first.point.parametricOffset)
203  {
204  drivingRange.minimum = origin.first.point.parametricOffset;
205  drivingRange.maximum = neighbor.first.point.parametricOffset;
206  }
207  else
208  {
209  drivingRange.minimum = neighbor.first.point.parametricOffset;
210  drivingRange.maximum = origin.first.point.parametricOffset;
211  }
212 
213  if (originLane == neighborLane)
214  {
215  neighborDuration = getDuration(*originLane, drivingRange);
216  }
217  else
218  {
219  auto const maxSpeed = std::max(getMaxSpeed(*originLane, drivingRange), cMinimumSpeed);
220  neighborDuration = neighborDistance / maxSpeed;
221  }
222  neighborDistance = std::max(neighborDistance, cMinimumNeighborDistance);
223  neighborDuration = std::max(neighborDuration, cMinimumNeighborDuration);
224  neighbor.second.routeDistance = origin.second.routeDistance + neighborDistance;
225  neighbor.second.routeDuration = origin.second.routeDuration + neighborDuration;
226  return neighbor;
227 }

242 void RouteExpander<RoutingCostData>::expandSameLaneNeighbors(
243  lane::Lane::ConstPtr lane, typename RouteExpander<RoutingCostData>::RoutingPoint const &origin)
244 {
245  if ((lane->id == getDest().laneId)
246  && ((isPositiveMovement(lane, origin) && (origin.first.point.parametricOffset <= getDest().parametricOffset))
247  || (isNegativeMovement(lane, origin) && (origin.first.point.parametricOffset >= getDest().parametricOffset))))
248  {
249  // approaching the destination from valid side
250  auto const neighbor = createNeighbor(lane, origin, lane, getRoutingDest());
251  addNeighbor(lane, origin, lane, neighbor, ExpandReason::Destination);
252  }
253  if (isPositiveMovement(lane, origin) && !isEnd(origin))
254  {
255  auto const neighbor = createNeighbor(
256  lane, origin, lane, createRoutingParaPoint(lane->id, physics::ParametricValue(1.), origin.first.direction));
257  addNeighbor(lane, origin, lane, neighbor, ExpandReason::SameLaneNeighbor);
258  }
259  if (isNegativeMovement(lane, origin) && !isStart(origin))
260  {
261  auto const neighbor = createNeighbor(
262  lane, origin, lane, createRoutingParaPoint(lane->id, physics::ParametricValue(0.), origin.first.direction));
263  addNeighbor(lane, origin, lane, neighbor, ExpandReason::SameLaneNeighbor);
264  }

267 template <class RoutingCostData>
268 void RouteExpander<RoutingCostData>::expandLongitudinalNeighbors(
269  lane::Lane::ConstPtr lane, typename RouteExpander<RoutingCostData>::RoutingPoint const &origin)
270 {
271  lane::ContactLaneList contactLanes;
272  if (isEnd(origin) && isPositiveMovement(lane, origin))
273  {
274  contactLanes = getContactLanes(*lane, lane::ContactLocation::SUCCESSOR);
275  }
276  else if (isStart(origin) && isNegativeMovement(lane, origin))
277  {
278  contactLanes = getContactLanes(*lane, lane::ContactLocation::PREDECESSOR);
279  }
280  for (auto contactLane : contactLanes)
281  {
282  lane::Lane::ConstPtr otherLane = lane::getLanePtr(contactLane.toLane);
283  if (otherLane)
284  {
285  if (isRouteable(*otherLane))
286  {
287  lane::ContactLocation otherToLane = getContactLocation(*otherLane, lane->id);
288  if (otherToLane == lane::ContactLocation::SUCCESSOR)
289  {
290  auto routingDirection = RoutingDirection::NEGATIVE;
291  if (origin.first.direction == RoutingDirection::DONT_CARE)
292  {
293  routingDirection = RoutingDirection::DONT_CARE;
294  }
295 
296  auto const neighbor = createLongitudinalNeighbor(
297  origin, createRoutingParaPoint(otherLane->id, physics::ParametricValue(1.), routingDirection));
298 
299  addNeighbor(lane, origin, otherLane, neighbor, ExpandReason::LongitudinalNeighbor);
300  }
301  else if (otherToLane == lane::ContactLocation::PREDECESSOR)
302  {
303  auto routingDirection = RoutingDirection::POSITIVE;
304  if (origin.first.direction == RoutingDirection::DONT_CARE)
305  {
306  routingDirection = RoutingDirection::DONT_CARE;
307  }
308  auto const neighbor = createLongitudinalNeighbor(
309  origin, createRoutingParaPoint(otherLane->id, physics::ParametricValue(0.), routingDirection));
310 
311  addNeighbor(lane, origin, otherLane, neighbor, ExpandReason::LongitudinalNeighbor);
312  }
313  else
314  {
315  throw std::runtime_error("Other lane neither SUCCESSOR not PREDECESSOR!");
316  }
317  }
318  }
319  else
320  {
321  throw std::runtime_error("No other lane!");
322  }
323  }
