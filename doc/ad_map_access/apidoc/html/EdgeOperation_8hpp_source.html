135 PointType getParametricPoint(std::vector<PointType> const &edge,
136  physics::Distance const &edgeLength,
137  const physics::ParametricValue &t)
138 {
139  if (!edge.empty())
140  {
141  physics::Distance length(0);
142  physics::Distance length_t = edgeLength * t;
143  for (size_t i = 0; i < edge.size() - 1; i++)
144  {
145  const auto &pt0 = edge[i];
146  const auto &pt1 = edge[i + 1];
147  auto d = distance(pt0, pt1);
148  if (d != physics::Distance(0.))
149  {
150  auto length_1 = length + d;
151  if (length_1 >= length_t)
152  {
153  auto d_t = length_t - length;
154  physics::ParametricValue tt(d_t / d);
155  return vectorInterpolate(pt0, pt1, tt);
156  }
157  else
158  {
159  length = length_1;
160  }
161  }
162  }

269 physics::ParametricValue
270 findNearestPointOnEdge(std::vector<PointType> const &edge, physics::Distance const &edgeLength, const PointType &pt)
271 {
272  if (isValid(pt))
273  {
274  if (edge.size() == 0)
275  {
276  return physics::ParametricValue();
277  }
278  else if ((edge.size() == 1) || (edgeLength == physics::Distance(0.)))
279  {
280  return physics::ParametricValue(0);
281  }
282  else
283  {
284  physics::ParametricValue t_one = findNearestPointOnSegment(pt, edge[0], edge[1]);
285  PointType pt_nearest = vectorInterpolate(edge[0], edge[1], t_one);
286  physics::Distance d_nearest = distance(pt, pt_nearest);
287  physics::Distance offset_nearest = distance(pt_nearest, edge[0]);
288  physics::Distance running_offset = physics::Distance(0);
289  for (size_t i = 1; i + 1 < edge.size(); i++)
290  {
291  physics::ParametricValue t = findNearestPointOnSegment(pt, edge[i], edge[i + 1]);
292  PointType pt_candidate = vectorInterpolate(edge[i], edge[i + 1], t);
293  physics::Distance d = distance(pt_candidate, pt);
294  running_offset += distance(edge[i - 1], edge[i]);
295  if (d < d_nearest)
296  {
297  pt_nearest = pt_candidate;
298  d_nearest = d;
299  offset_nearest = running_offset + distance(pt_nearest, edge[i]);
300  }
301  }
302  return physics::ParametricValue(offset_nearest / edgeLength);
303  }
304  }
305  else
306  {
307  return physics::ParametricValue();
308  }

336 template <typename PointType>
337 std::pair<physics::MetricRange, physics::Distance> calculateWidthRange(std::vector<PointType> const &edgeLeft,
338  physics::Distance const &edgeLeftLength,
339  std::vector<PointType> const &edgeRight,
340  physics::Distance const &edgeRightLength)
341 {
342  physics::MetricRange widthRange;
343  widthRange.minimum = std::numeric_limits<physics::Distance>::max();
344  widthRange.maximum = physics::Distance(0.);
345  physics::Distance widthSum(0.);
346  size_t widthSumCount(0u);
347  physics::ParametricValue parametricStepsize(0.5);
348  auto length = (edgeLeftLength + edgeRightLength) * 0.5;
349  if (length > physics::Distance(1.))
350  {
351  parametricStepsize
352  = std::max(physics::ParametricValue(0.01), physics::ParametricValue(1. / static_cast<double>(length)));
353  }
354  physics::ParametricValue longitudinalOffset(0.);
355  bool endReached = false;
356  do
357  {
358  auto const pt0 = getParametricPoint(edgeLeft, edgeLeftLength, longitudinalOffset);
359  auto const pt1 = getParametricPoint(edgeRight, edgeRightLength, longitudinalOffset);
360  if (!isValid(pt1) || !isValid(pt1))
361  {
362  return std::make_pair(physics::MetricRange(), physics::Distance());
363  }
364  auto const centerPoint = point::vectorInterpolate(pt0, pt1, physics::ParametricValue(0.5));
365  if (!isValid(centerPoint))
366  {
367  return std::make_pair(physics::MetricRange(), physics::Distance());
368  }
369  auto const longTLeft = findNearestPointOnEdge(edgeLeft, edgeLeftLength, centerPoint);
370  auto const longTRight = findNearestPointOnEdge(edgeRight, edgeRightLength, centerPoint);
371  if (!isRangeValid(longTLeft) || !isRangeValid(longTRight))
372  {
373  return std::make_pair(physics::MetricRange(), physics::Distance());
374  }
375  auto const pointOnLeftEdge = getParametricPoint(edgeLeft, edgeLeftLength, longTLeft);
376  auto const pointOnRightEdge = getParametricPoint(edgeRight, edgeRightLength, longTRight);
377  if (!isValid(pointOnLeftEdge) || !isValid(pointOnRightEdge))
378  {
379  return std::make_pair(physics::MetricRange(), physics::Distance());
380  }
381  physics::Distance const width = distance(pointOnLeftEdge, pointOnRightEdge);
382  unionRangeWith(widthRange, width);
383  widthSum += width;
384  widthSumCount++;
385  if (longitudinalOffset < physics::ParametricValue(1.))
386  {
387  longitudinalOffset += parametricStepsize;
388  if (longitudinalOffset > physics::ParametricValue(1.))
389  {
390  longitudinalOffset = physics::ParametricValue(1.);
391  }
392  }
393  else
394  {
395  endReached = true;
396  }
397  } while (!endReached);
398 
399  auto averageWidth = (widthSum / static_cast<double>(widthSumCount));
400 
401  return std::make_pair(widthRange, averageWidth);
402 }
