192 std::vector<PointType> getParametricRange(std::vector<PointType> const &edge,
193  physics::Distance const &edgeLength,
194  const physics::ParametricRange &trange)
195 {
196  typedef std::vector<PointType> EdgeType;
197  if (!edge.empty())
198  {
199  EdgeType pts;
200  physics::Distance length(0);
201  physics::Distance length_t_start = edgeLength * trange.minimum;
202  size_t i = 0;
203  for (; i < edge.size() - 1 && pts.empty(); i++)
204  {
205  const PointType &pt0 = edge[i];
206  const PointType &pt1 = edge[i + 1];
207  physics::Distance d = distance(pt0, pt1);
208  physics::Distance length_1 = length + d;
209  if (length_1 >= length_t_start)
210  {
211  if (d != physics::Distance(0))
212  {
213  physics::Distance d_t = length_t_start - length;
214  physics::ParametricValue tt(d_t / d);
215  PointType pt_start = vectorInterpolate(pt0, pt1, tt);
216  pts.push_back(pt_start);
217  }
218  else
219  {
220  pts.push_back(pt1);
221  }
222  // ensure i++ is not executed in this case
223  break;
224  }
225  else
226  {
227  length = length_1;
228  }
229  }
230  if (pts.empty())
231  {
232  // this can happen if the length was measured e.g. in ECEF and the cumulated
233  // length of the ENU cached lane coordinates are slightly smaller
234  // and the range is near to the end of the lane
235  pts.push_back(edge.back());
236  }
237  physics::Distance length_t_end = edgeLength * trange.maximum;
238  for (; i < edge.size() - 1; i++)
239  {
240  const PointType &pt0 = edge[i];
241  const PointType &pt1 = edge[i + 1];
242  physics::Distance d = distance(pt0, pt1);
243  physics::Distance length_1 = length + d;
244  if (length_1 >= length_t_end)
245  {
246  if (d != physics::Distance(0))
247  {
248  physics::Distance d_t = length_t_end - length;
249  physics::ParametricValue tt(d_t / d);
250  PointType pt_end = vectorInterpolate(pt0, pt1, tt);
251  pts.push_back(pt_end);
252  }
253  else
254  {
255  pts.push_back(pt1);
256  }
257  return pts;
258  }
259  else
260  {
261  pts.push_back(pt1);
262  length = length_1;
263  }
264  }
265  pts.push_back(edge.back());
266  return pts;
267  }
268  return EdgeType();

290 physics::ParametricValue
291 findNearestPointOnEdge(std::vector<PointType> const &edge, physics::Distance const &edgeLength, const PointType &pt)
292 {
293  if (isValid(pt))
294  {
295  if (edge.size() == 0)
296  {
297  return physics::ParametricValue();
298  }
299  else if ((edge.size() == 1) || (edgeLength == physics::Distance(0.)))
300  {
301  return physics::ParametricValue(0);
302  }
303  else
304  {
305  physics::ParametricValue t_one = findNearestPointOnSegment(pt, edge[0], edge[1]);
306  PointType pt_nearest = vectorInterpolate(edge[0], edge[1], t_one);
307  physics::Distance d_nearest = distance(pt, pt_nearest);
308  physics::Distance offset_nearest = distance(pt_nearest, edge[0]);
309  physics::Distance running_offset = physics::Distance(0);
310  for (size_t i = 1; i + 1 < edge.size(); i++)
311  {
312  physics::ParametricValue t = findNearestPointOnSegment(pt, edge[i], edge[i + 1]);
313  PointType pt_candidate = vectorInterpolate(edge[i], edge[i + 1], t);
314  physics::Distance d = distance(pt_candidate, pt);
315  running_offset += distance(edge[i - 1], edge[i]);
316  if (d < d_nearest)
317  {
318  pt_nearest = pt_candidate;
319  d_nearest = d;
320  offset_nearest = running_offset + distance(pt_nearest, edge[i]);
321  }
322  }
323  return physics::ParametricValue(offset_nearest / edgeLength);
324  }
325  }
326  else
327  {
328  return physics::ParametricValue();
329  }

357 template <typename PointType>
358 std::pair<physics::MetricRange, physics::Distance> calculateWidthRange(std::vector<PointType> const &edge_left,
359  physics::Distance const &edgeLeftLength,
360  std::vector<PointType> const &edge_right,
361  physics::Distance const &edgeRightLength)
362 {
363  physics::MetricRange width_range;
364  width_range.minimum = std::numeric_limits<physics::Distance>::max();
365  width_range.maximum = physics::Distance(0.);
366  physics::Distance widthSum(0.);
367  size_t widthSumCount(0u);
368  physics::ParametricValue parametricStepsize(0.5);
369  auto length = (edgeLeftLength + edgeRightLength) * 0.5;
370  if (length > physics::Distance(1.))
371  {
372  parametricStepsize = std::max(physics::ParametricValue(0.01), physics::ParametricValue(1. / length.mDistance));
373  }
374  physics::ParametricValue longitudinalOffset(0.);
375  bool endReached = false;
376  do
377  {
378  auto const pt0 = getParametricPoint(edge_left, edgeLeftLength, longitudinalOffset);
379  auto const pt1 = getParametricPoint(edge_right, edgeRightLength, longitudinalOffset);
380  if (!isValid(pt1) || !isValid(pt1))
381  {
382  return std::make_pair(physics::MetricRange(), physics::Distance());
383  }
384  auto const center_point = point::vectorInterpolate(pt0, pt1, physics::ParametricValue(0.5));
385  if (!isValid(center_point))
386  {
387  return std::make_pair(physics::MetricRange(), physics::Distance());
388  }
389  auto const longTLeft = findNearestPointOnEdge(edge_left, edgeLeftLength, center_point);
390  auto const longTRight = findNearestPointOnEdge(edge_right, edgeRightLength, center_point);
391  if (!isRangeValid(longTLeft) || !isRangeValid(longTRight))
392  {
393  return std::make_pair(physics::MetricRange(), physics::Distance());
394  }
395  auto const pointOnLeftEdge = getParametricPoint(edge_left, edgeLeftLength, longTLeft);
396  auto const pointOnRightEdge = getParametricPoint(edge_right, edgeRightLength, longTRight);
397  if (!isValid(pointOnLeftEdge) || !isValid(pointOnRightEdge))
398  {
399  return std::make_pair(physics::MetricRange(), physics::Distance());
400  }
401  physics::Distance const width = distance(pointOnLeftEdge, pointOnRightEdge);
402  unionRangeWith(width_range, width);
403  widthSum += width;
404  widthSumCount++;
405  if (longitudinalOffset < physics::ParametricValue(1.))
406  {
407  longitudinalOffset += parametricStepsize;
408  if (longitudinalOffset > physics::ParametricValue(1.))
409  {
410  longitudinalOffset = physics::ParametricValue(1.);
411  }
412  }
413  else
414  {
415  endReached = true;
416  }
417  } while (!endReached);
418 
419  auto averageWidth = (widthSum / static_cast<double>(widthSumCount));
420 
421  return std::make_pair(width_range, averageWidth);
422 }
